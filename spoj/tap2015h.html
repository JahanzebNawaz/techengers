    #include<bits/stdc++.h>
    #include<algorithm>
    using namespace std;
     
    int main()
    {
    	long int n;
    	while(scanf("%ld",&n)!=EOF)
    	{
    		long int num=0,q,i,x=0,a[100],cnt=0;
    		while(n!=0)
    		{
    			q=n;
    			while(q!=0)
    			{
    				a[x++]=q%10;
    				q/=10;
    			}
    			sort(a,a+x);
    			for(i=0;i<x;i++)
    			{
    				num = num*10 + a[i];
    			}
    			n=n-num;
    			//cout<<n<<"here"<<"\n";
    			cnt++;
    			x=0;
    			num=0;
    		}
    		cout<<cnt<<endl;
    	}
    	return 0;
    }
