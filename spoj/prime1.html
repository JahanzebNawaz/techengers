#include <stdio.h>
#include<math.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int m,n;
		scanf("%lld %lld",&m,&n);
		long int i,j,flag=0;
		if(m==1)
		m++;
		for(i=m;i<=n;i++)
		{
			if(i%2==0&&m!=2)
			continue;
			for(j=2;j<=sqrt(i);j++)
			{
				if(i%j==0)
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			printf("%ld\n",i);
			flag=0;
		}
		printf("\n");
	}
	return 0;
}
