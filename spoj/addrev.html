#include<stdio.h>
int main()
{
	int a[10000]={0},i,j;
	for(i=2;i<10000;i++)
	{   
		if(a[i]==0){
		for(j=i+1;j<10000;j++)
		{
			if(j%i == 0)
			{
				a[j]++;
			}
		}}
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		for(i=30;i<10000;i++)
		{
			if(a[i]>=3)
			{
			
				n--;
				if(n==0)
				{
					printf("%d\n",i);
					break;
				}
				//	i++;
			}
		}
	}
	return 0;
}
