HACKRNDM - Hacking the random number generator


  • C/C++
    #include <bits/stdc++.h>
    #include <algorithm>
    using namespace std;
     
    int bsearch( int array[], int num, int N ) {
    	int beg = 0, end = N;
     
    	while ( beg <= end ) {
    		int mid = ( beg + end ) / 2;
     
    		if ( array[ mid ] > num ) 
    		{
    			end = mid - 1;
    		}
    		else if ( array[ mid ] < num ) 
    		{
    			beg = mid + 1;
    		}
    		else 
    		{
    			return 1;
    		}
    	}
    	return 0;
    }
     
    int main()
    {
    	int n, k;
    	scanf( "%d %d", &n, &k );
    	int a[n], i, count = 0;
     
    	for ( i = 0; i < n; ++i ) {
    		scanf( "%d",&a[i]);
    	}
     
    	sort( a, a + n );
     
    	for ( i = 0; i < n; ++i ) {
    		if ( bsearch( a, a[ i ] + k, n ) ) {
    			++count;
    		}
    	}
     
    	printf( "%d\n", count );
    	return 0;
    }
