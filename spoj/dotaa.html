using namespace std;

int main()
{
    int t,n,i,m,d,h[500],count=0;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d",&n,&m,&d);
        count=0;
 
        for(i=0;i<n;i++)
            {
                scanf("%d",&h[i]);
                while((h[i]-=d)>0) 
                {
                count++;}
            }
            
          if(count>=m)
          printf("YES\n");
          else 
          printf("NO\n");
    }
    return 0;
}
