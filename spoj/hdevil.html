#include<math.h>
int main(void) {
	int t,e=1;
	scanf("%d",&t);
	while(t--)
	{
		long long int m,n,x,y,a,b,i,sum=0,j,count=0;
		scanf("%lld%lld",&n,&m);
		x=sqrt(n);
		for(i=2;i<=x;i++)
		{
			if(n%i==0)
			{
				if(i*i==n)
				sum=sum+i;	
				else
				sum=sum+i+(n/i);	
			}	
		}
			sum=sum+1;
			sum=sum%m;
			for(j=2;j<=sqrt(sum);j++)
			{
				if(sum%j==0){
				count+=2;
				if(j==sum/j)
				count--;}
			}count++;
			x=(5*count*count)+4;
			y=(5*count*count)-4;
			a=sqrt(x);
			b=sqrt(y);
			if((a*a==x)||(b*b==y))
			printf("Case #%d : YES.\n",e);
			else
			printf("Case #%d : NO.\n",e);	
			e++;
	}
	return 0;
}
