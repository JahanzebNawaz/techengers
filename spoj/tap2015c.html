    #include <bits/stdc++.h>
    #include<string.h>
    #include<algorithm>
    using namespace std;
     
    int main()
    {
    	ios::sync_with_stdio(false); cin.tie(0);
    	int n,s,j,d;
    	while(scanf("%d %d %d %d",&n,&s,&j,&d)!=EOF)
    	{
    		char a[100009];
    		scanf("%s",a);
    //		cout<<a;
    //		cout<<a;
    		int l,i,ca=0,cb=0,sa=0,sb=0;
    		l=strlen(a);
    		for(i=0;i<l;i++)
    		{
    			if(ca!=j && cb!=j)
    			{
    				if(a[i]=='A')
    					ca++;
    				else
    					cb++;
    			}
    			if(ca==j || cb==j)
    			{
    				while((ca-cb)<d && (cb-ca)<d && i<l)
    				{
    					i++;
    					if(a[i]=='A')
    						ca++;
    					else
    						cb++;
    				}
    				if((ca-cb)>=d)
    				{
    					sa++;
    					ca=0;
    					cb=0;
    				}
    				if((cb-ca)>=d)
    				{
    					sb++;
    					ca=0;
    					cb=0;
    				}
    			}
    		}
    		cout<<sa<<" "<<sb<<"\n";
    	}
    	return 0;
    }
