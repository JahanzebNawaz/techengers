ABSURD - Absurd prices


  • C/C++
    #include <iostream>
    #include<cmath>
    using namespace std;
     
    int main() {
    	int t;
    	cin.sync_with_stdio(false);
    	cin>>t;
    	while(t--)
    	{
    		long long int n,absurd,a,d,x,x1,x2,i,flag=0,absurd1,c;
    		cin>>n;
    		x=n;
    		while(n%10==0)
    		{
    			n/=10;
    		}
    		d=n%10;
    		a=floor(log10((double)n))+1;
    		if(d==5)
    		absurd=2*a-1;
    		else absurd=2*a;
    		x1=ceil(.95*n);x2=floor(1.05*n);
    		for(i=x1;i<=x2;i++)
    		{
    			c=i;
    		    while(c%10==0)
    		  {
    			c/=10;
    		  }
    		  a=floor(log10((double)c))+1;
    		  d=c%10;
    		  if(d==5)
    		absurd1=2*a-1;
    		else absurd1=2*a;
    		if(absurd>absurd1)
    		{
    			flag=1;
    			break;
    		}
    		}
    		if(flag==1)
    		cout<<"absurd"<<endl;
    		else
    		cout<<"not absurd"<<endl;
    	}
    	return 0;
    }
