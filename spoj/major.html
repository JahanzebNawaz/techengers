    #include<algorithm>
    using namespace std;
    int main()
    {
    	int t;
    	scanf("%d",&t);
    	while(t--)
    	{
    		int max=0;
    		int n;
    		scanf("%d",&n);
    		int a[n],i,k,count=0;
    		for(i=0;i<n;i++)
    		scanf("%d",&a[i]);
    		sort(a,a+n);
    		for(i=0;i<n;i++)
    		{
    			if(a[i]==a[i+1])
    			{
    				count++;
    				
    		   }
    		    if(count>max)
    			 {
    			 	max=count;
    			 	k=a[i];
    			 }
    			if(a[i]!=a[i+1])
    			count=0;
    		}
    		if((max+1)>n/2)
    		printf("YES %d\n",k);
    		else 
    		printf("NO\n");
    	}
    	return 0;
    }
