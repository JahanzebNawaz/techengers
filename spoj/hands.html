		else:

	max = list(max)
	a = [0,106,211,317,422,528,633,739,844,950,1055,1200,1306,1411,1517,1622,1728,1833,1939,2044,2150,2255,2400]
	min.remove(':')

	c = 0
	l = int(''.join(min))
	m = int(''.join(max))
	for w in a:
		if w>l and w<=m:
			c = c + 1
	print c
