AMR10G - Christmas Play


  • C/C++
    #include<iostream>
    #include<cstdio>
    #include<algorithm>
    long long int a[10000000],b[10000000];
    using namespace std;
    int main()
    {
    	int t;
    	scanf("%d",&t);
    	while(t--)
    	{
    		int n,k,x;
    		scanf("%d %d",&n,&k);
    		x=k;
    		long long int i,m=0;
    		for(i=0;i<n;i++)
    		scanf("%lld",&a[i]);
    		if(k==1)
    		printf("0\n");
    		else 
    		{
    			sort(a,a+n);
    			for(i=0;i<=(n-k);i++)
    			{
    				b[m]=(a[x-1]-a[i]);
    				m++;
    				x++;
    			}
    			sort(b,b+(m));
    			printf("%lld\n",b[0]);
    		}
    	}
    	return 0;
    }
