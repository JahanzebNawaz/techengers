    #include <iostream>
    #include<algorithm>
    using namespace std;
    int main()
    {
    	int t;
    	cin.sync_with_stdio(false);
    	cin>>t;
    	while(t--)
    	{
    		int n,m;
    		cin>>n>>m;
    		int a[n],i,low,high,count=0;
    		for(i=0;i<n;i++)
    		cin>>a[i];
    		sort(a,a+n);
    		low=0;
    		high=n-1;
    		while(high>low)
    		{
    			if(a[low]+a[high]>m)
    			high--;
    			else if(a[low]+a[high]<m)
    			low++;
    			else if(a[low]+a[high]==m)
    			{
    				count++;
    				high--;
    				low++;
    			}
    		}
    		cout<<count<<endl;
    	}
    	return 0;
    }
