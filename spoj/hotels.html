        scanf("%lld%lld",&n, &m);
            long a[n];
            for(i=0; i<n; i++)
            {
                 scanf("%d", &a[i]);
            }
            j=0;
            max = 0;
            sum = 0;
            for(i=0; i<n; i++)
            {
                sum = sum + a[i];
                while(sum>m)
                {
                    sum = sum - a[j];
                    j++;
                    if(max<=sum && sum<=m)
                    {
                        max = sum;
                        break;
                    }
                }
                if(max<=sum && sum<=m)
                {
                    max = sum;
                }
            }
            printf("%lld\n", max);
        
    }
