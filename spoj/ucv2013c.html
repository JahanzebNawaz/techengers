#include<cmath>

#include <vector>

#include <string.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;

int main()
{
    long long int d,f,b,m,temp,diff,mon;
    cin>>d>>f>>b>>m;
    while(!(d==0 && f==0 && b==0 && m==0))
    {
        temp=(f*(f+1))/2+1;
        mon=b*temp;
        diff=mon+m-d;
        if(diff>0)
            printf("The firm is trying to bankrupt Farmer Cream by %lld Bsf.\n",diff);
        else
            printf("Farmer Cream will have %lld Bsf to spend.\n",d-mon);
        cin>>d>>f>>b>>m;
    }
    return 0;
}
