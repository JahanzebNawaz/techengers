    long double bs(long double array[])
    {
        long double ini=0,last=array[n-1],max=0.0;
        while (last - ini >= 1e-6)
        {
            long double mid=(ini+last)/2;
            if (func(mid,array)==1)
            {
                ini=mid;
            }
            else
                last=mid;
        }
        return ini;
    }
