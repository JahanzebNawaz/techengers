     a=sqrt(1-4*k);
     a=a/2;
     x=sqrt(0.5+a);
     ang=asin(x);
     res=(k1*((u*u*2*x*sqrt(1-x*x))/10)+((k2*u*u*x*x)/(2*10)));
     printf("%0.3lf %0.3lf\n",ang,res);
     }
     return 0;
    }
