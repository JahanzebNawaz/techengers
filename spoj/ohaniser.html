    #define ll long long
    #define mod 1000000007
    ll modular_pow(ll base, ll exponent, int modulus)
    {
        ll result = 1;
        while (exponent > 0)
        {
            if (exponent % 2 == 1)
                result = (result * base) % modulus;
            exponent = exponent >> 1;
            base = (base * base) % modulus;
        }
        return result;
    }
