    #include<stdio.h>
    int main()
    {
    	int t;
    	scanf("%d",&t);
    	while(t--)
    	{
    		long long n,i;
    		double ans=1.5;
    		scanf("%lld",&n);
    		if(n==1)
    		printf("1.0\n");
    		else if(n==2)
    		printf("1.5\n");
    		else
    		{
    			for(i=3;i<=n;i++)
    			{
    				ans=ans+(1.0)/i;
    			}
    			printf("%.6lf\n",ans);
    		}
    	}
    	return 0;
    }
