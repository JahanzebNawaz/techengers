int josephus(int n, int k)
{
  if (n == 1)
    return 1;
  else
    return (josephus(n - 1, k+1) + k-1) % n + 1;
}
 
int main()
{
	int ans,n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=josephus(n,1);
		cout<<ans<<endl;
	}
	return 0;
}
