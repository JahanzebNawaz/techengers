GAMES - HOW MANY GAMES


  • C/C++
#include <bits/stdc++.h>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
int main(void) 
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string s;
		cin>>s;
		int i,j,count=0,frac=0,a,b,temp,flag=0,x;
		j=s.length();
		for(i=j-1;i>=0;i--)
		{
			if(s[i]=='.')
			{ flag=1;
			break;}
			else
			{count++;
			frac=(s[i]-'0')*(pow(10,count-1))+frac;}
		}
	//	printf("%d\n",count);
		if(flag==0)
		printf("1\n");
		else{
		a=pow(10,count);
		x=a;
		b=frac;
		while(b!=0)
		{
			temp=b;
			b=a%b;
			a=temp;
		}
		x=x/a;
		printf("%d\n",x);}
	}
	return 0;
}
