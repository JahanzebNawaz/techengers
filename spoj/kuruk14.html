     #include <iostream>
    using namespace std;
     
    int main()
    {
    	ios::sync_with_stdio(false);
    	cin.tie(0);
    	int n,t,m,i;
    	cin>>t;
    	while(t--)
    	{
    		int cnt=0,a[1010]={0};
    		cin>>n;
    		for(i=0;i<n;i++)
    		{
    			cin>>m;
    			if(a[m+1]==0)
    			{
    				a[m+1]=1;
    				cnt++;
    			}
    			else if(a[n-m]==0)
    			{
    				a[n-m]=1;
    				cnt++;
    			}
    		}
    		if(cnt==n)
    			cout<<"YES"<<endl;
    		else
    			cout<<"NO"<<endl;
    	}
    	return 0;
    }
