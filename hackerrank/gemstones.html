#include<cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
	int n,i,j,ctnum=0,ans=0;
	cin>>n;
	cin.ignore(254,'\n');                           
	int ct[n][26];
	char str[n][110];
	for(i=0;i<n;i++)
	{
		for(j=0;j<26;j++)
		{
			ct[i][j]=0;
		}
	}
	for(i=0;i<n;i++)
	{
		gets(str[i]);
	}

	
	
	for(i=0;i<n;i++)
	{
		for(j=0;str[i][j]!='\0';j++)
		{
			//cout<<"here";
			ct[i][str[i][j]-'a']=1;
		}
	}

for(i=0;i<26;i++)
    {
	ctnum=0;
		for(j=0;j<n;j++)
	{
	if(ct[j][i]==1)
	     {
		ctnum++;
	     }
	}
	if(ctnum==n)
	{
		ans++;
	}
    }
	
	cout<<ans<<"\n";
	return 0;

}
