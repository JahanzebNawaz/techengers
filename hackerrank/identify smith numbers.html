#include<cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;


int main()    
{
	unsigned long int n,cn,sum=0,i,snum=0,d,ci,rem;
	
	cin>>n;
	cn=n;
	for(i=2;i<=n;i++)
	{
		while(n%i==0)
		{
			ci=i;
			//cout<<i<<" ";
			if((n/10)!=0)
			{
				while(ci>0)
				{
					rem=ci%10;
					ci=ci/10;
					sum=sum+rem;
				}
			}
			else
			{
			  sum=sum+i;
	     	}
			n=n/i;
		}
		
		if(n==0)
		{
			break;
		}
	}
	//cout<<sum<<"\n";
	i=0;
	while(cn>0)
	{
		d=cn%10;
		cn=cn/10;
		snum=snum+d;
		
	}
	
	if(snum==sum)
	{
		cout<<"1\n";
	}
	else
	{
		cout<<"0\n";
	}
	
	return 0;
}
