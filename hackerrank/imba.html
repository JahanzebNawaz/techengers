#include<cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;                             

int main() {
	int t;
	cin>>t;
	while(t--)
	{
	long long int i,j,n;
	cin>>n;
	long long int arr[n],k;
		if(n%2==0)
		{
			k=0;
			for(i=n/2;i>=1;i--)
			{
				arr[k]=i;
				k=k+2;
			}
			k=1;
			for(i=n/2+1;i<=n;i++)
			{
				arr[k]=i;
				k=k+2;
			}
		}
		
		else
		{	
			k=0;
			for(i=n/2+1;i<=n;i++)
			{
				arr[k]=i;
				k=k+2;
			}
			k=1;
			for(i=n/2;i>=1;i--)
			{
				arr[k]=i;
				k=k+2;
			}
		}
		
		for(i=0;i<n;i++)
		{
			cout<<arr[i]<<" ";
		}
		cout<<"\n";
	}
	return 0;
}
