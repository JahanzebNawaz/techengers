SHERLOCK AND THE BEAST

LINK:https://www.hackerrank.com/challenges/sherlock-and-the-beast


  • C++
#include<cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,m,i,k,flag=0;
		cin>>n;
		m=3*(n/3);
		k=n%3;
		if(n%3==0)
		{
			for(i=0;i<n;i++)
			cout<<"5";
			flag=1;
		}
		else
		{
			
			
			while(m>=0)
			{
				if(m%3==0&&k%5==0)
				{
					//cout<<k;
					for(i=0;i<m;i++)
				cout<<"5";
					for(i=0;i<k;i++)
					cout<<"3";
					flag=1;
					break;
				}
				m-=3;
				k+=3;
			}
		}
		
		if(flag==0)
		cout<<"-1";
		cout<<endl;
	}
	
	
	return 0;
}
