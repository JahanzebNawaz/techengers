#include <algorithm>
using namespace std;

 int height(int n) {
    int k,ht=1;
    for(k=1;k<=n;k++)
        {
        if(k%2!=0)
            {
            ht=2*ht;
        }
        else
            {
            ht=ht+1;
        }
         
    }
    return ht;
    
}
int main() {
    int T;
    cin >> T;
    while (T--) {
        int n,x;
        cin >> n;
        if(n==0)
            {
            cout<<"1"<<"\n";
            continue;
        }
        x=height(n);
        cout<<x<<"\n";
    }
    cout<<"\n";
    return 0;
}
