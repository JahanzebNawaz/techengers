#include<cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;


int main() {

	unsigned long long int ct;
	int len,t,o,te,th,h,i;
	
	cin>>t;
	cin.ignore(254,'\n');
	char str[1100],thous[1100];
	while(t--)
	{
		ct=0;
		for(i=0;i<1100;i++)
	    {
		thous[i]='\0';
		str[i]='\0';
	    }
		gets(str);
		len=strlen(str);
	char one[4],ten[4],hun[4];
	for(i=0;i<=4;i++)
	    {
		one[i]='\0';
		ten[i]='\0';
		hun[i]='\0';
	    }
	
		for(i=len;i>=0;i--)
		{
			if(str[i]=='I')
			{
				ct=ct+1;
			}
			if(str[i]=='V')
			{
				ct=ct+5;
			}
			if(str[i]=='X')
			{
				ct=ct+10;
			}
			if(str[i]=='L')
			{
				ct=ct+50;
			}
			if(str[i]=='C')
			{
				ct=ct+100;
			}
			if(str[i]=='D')
			{
				ct=ct+500;
			}
			if(str[i]=='M')
			{
				ct=ct+1000;
			}
		}
		
		o = ct%10;
		ct=ct/10;
		te = (ct%10)*10;
		ct=ct/10;
		h = (ct%10)*100;
		ct=ct/10;
		th = ct;
		
		
		if(o==1)
		{
			one[0]='I';
		}
		
		else if(o==2)
		{
			one[0]='I';
			one[1]='I';
		}
		else if(o==3)
		{
			one[0]='I';
			one[1]='I';
			one[2]='III';
		}
		else if(o==4)
		{
			one[0]='I';
			one[1]='V';
		}
		else if(o==5)
		{
			one[0]='V';
			
		}
		else if(o==6)
		{
			one[0]='V';
			one[1]='I';
		}
		else if(o==7)
		{
		
			one[0]='V';
			one[1]='I';
			one[2]='I';
		}
		else if(o==8)
		{
			one[0]='V';
			one[1]='I';
			one[2]='I';
			one[3]='I';
		}
		else if(o==9)
		{
		one[0]='I';
		one[1]='X';
		}
		
		
		
		if(te==10)
		{
			ten[0]='X';
		}
		
		else if(te==20)
		{
			ten[0]='X';
			ten[1]='X';
		}
		else if(te==30)
		{
		ten[0]='X';
			ten[1]='X';
			ten[2]='X';
		}
		else if(te==40)
		{
		ten[0]='X';
		ten[1]='L';
		}
		else if(te==50)
		{
			ten[0]='L';
			
		}
		else if(te==60)
		{
			ten[0]='L';
			ten[1]='X';
		}
		else if(te==70)
		{
		ten[0]='L';
			ten[1]='X';
			ten[2]='X';
		}
		else if(te==80)
		{
			ten[0]='L';
			ten[1]='X';
			ten[2]='X';
			ten[3]='X';
		}
		else if(te==90)
		{
		ten[0]='X';
		ten[1]='C';
		
		}
		
	
		if(h==100)
		{
			hun[0]='C';
		}
		
		else if(h==200)
		{
			hun[0]='C';
			hun[1]='C';
		}
		else if(h==300)
		{
			hun[0]='C';
			hun[1]='C';
			hun[2]='C';
		}
		else if(h==400)
		{
			hun[0]='C';
			hun[1]='D';
		}
		else if(h==500)
		{
			hun[0]='D';
			
		}
		else if(h==600)
		{
			hun[0]='D';
			hun[1]='C';
		}
		else if(h==700)
		{
		hun[0]='D';
			hun[1]='C';
			hun[2]='C';
		}
		else if(h==800)
		{
			hun[0]='D';
			hun[1]='C';
			hun[2]='C';
			hun[3]='C';
		}
		else if(h==900)
		{
		hun[0]='C';
		hun[1]='M';
		}
		
		for(i=0;i<th;i++)
		{
			thous[i]='M';
		}
		
	strcat(thous,hun);
	strcat(thous,ten);
	strcat(thous,one);
	puts(thous);
		
		
	}
	
return 0;
}
